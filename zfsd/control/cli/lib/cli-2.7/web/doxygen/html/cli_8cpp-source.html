00133 const int Cli::FindFromName(Cli::List& CLI_CliList, const char* const STR_RegExp)
00134 {
00135 #ifndef CLI_NO_REGEX
00136     // Compile the regular expression.
00137     regex_t s_RegExp;
00138     memset(& s_RegExp, '\0', sizeof(regex_t));
00139     if (regcomp(& s_RegExp, STR_RegExp, 0) != 0)
00140     {
00141         return -1;
00142     }
00143 #endif
00144 
00145     // Check the name for each CLI instance.
00146     const int i_InitialCount = CLI_CliList.GetCount();
00147     for (   Cli::List::Iterator it = GetCliRegistry().GetIterator();
00148             GetCliRegistry().IsValid(it);
00149             GetCliRegistry().MoveNext(it))
00150     {
00151         if (const Cli* const pcli_Cli = GetCliRegistry().GetAt(it))
00152         {
00153 #ifndef CLI_NO_REGEX
00154             if (regexec(& s_RegExp, pcli_Cli->GetName(), 0, NULL, 0) == 0)
00155 #endif
00156             {
00157                 if (! CLI_CliList.AddTail(pcli_Cli))
00158                 {
00159                     CLI_ASSERT(false);
00160                 }
00161             }
00162         }
00163     }
00164 
00165 #ifndef CLI_NO_REGEX

00167     regfree(& s_RegExp);
