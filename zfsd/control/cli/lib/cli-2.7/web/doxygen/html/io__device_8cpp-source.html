00266 const OutputDevice& OutputDevice::operator <<(const double D_Out) const
00267 {
00268     // First of all, find out the appropriate format.
00269     char str_Format[128];
00270     if ((D_Out != 0.0) && (-1e-6 <= D_Out) && (D_Out <= 1e-6))
00271     {
00272         snprintf(str_Format, sizeof(str_Format), "%s", "%f");
00273     }
00274     else
00275     {
00276         int i_Decimal, i_Out;
00277         for (   i_Decimal = 6, i_Out = (int) floor(D_Out * 1e6 + 0.5); // floor(D+1/2) should be equivalent to round(D)
00278                 i_Decimal > 1;
00279                 i_Decimal --, i_Out /= 10)
00280         {
00281             if ((i_Out % 10) != 0)
00282             {
00283                 break;
00284             }
00285         }
00286         snprintf(str_Format, sizeof(str_Format), "%s.%d%s", "%", i_Decimal, "f");
00287     }
00288 
00289     // Format the float number with the computed format.
00290     char str_Out[128];
00291     snprintf(str_Out, sizeof(str_Out), str_Format, D_Out);
00292 
00293     // Output the computed string.
00294     PutString(str_Out);
00295 
00296     // Return the instance itself.
00297     return *this;
00298 }
