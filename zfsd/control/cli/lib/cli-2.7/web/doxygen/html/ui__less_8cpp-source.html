00081         void Less::OnKey(const KEY E_KeyCode)
00082         {
00083             // Ensure m_puiTextIt is valid.
00084             CLI_ASSERT(m_puiTextIt != NULL);
00085             if (m_puiTextIt == NULL) { Quit(); }
00086             else {
00087                 switch (E_KeyCode)
00088                 {
00089                 case KEY_BEGIN:
00090                     m_uiText.Begin(*m_puiTextIt);
00091                     PrintScreen();
00092                     break;
00093                 case PAGE_UP:
00094                     if (m_uiText.PageUp(*m_puiTextIt)) PrintScreen();
00095                     else GetShell().Beep();
00096                     break;
00097                 case KEY_UP:
00098                     if (m_uiText.LineUp(*m_puiTextIt)) PrintScreen();
00099                     else GetShell().Beep();
00100                     break;
00101                 case KEY_DOWN:
00102                 case ENTER:
00103                     if (m_uiText.LineDown(*m_puiTextIt, NULL)) PrintScreen();
00104                     else GetShell().Beep();
00105                     break;
00106                 case PAGE_DOWN:
00107                 case SPACE:
00108                     if (m_uiText.PageDown(*m_puiTextIt, NULL)) PrintScreen();
00109                     else GetShell().Beep();
00110                     break;
00111                 case KEY_END:
00112                     m_uiText.End(*m_puiTextIt, NULL);
00113                     PrintScreen();
00114                     break;
00115                 case KEY_q:
00116                 case KEY_Q:
00117                 case ESCAPE:
00118                 case BREAK:
00119                 case LOGOUT:
00120                 case NULL_KEY:
00121                     // Stop display
00122                     Quit();
00123                     break;
00124                 default:
00125                     // Non managed character: beep.
00126                     GetShell().Beep();
00127                     break;
00128                 }
00129             }
00130         }

00132         void Less::PrintScreen(void)
00133         {
00134             const OutputDevice::ScreenInfo cli_ScreenInfo = GetShell().GetStream(OUTPUT_STREAM).GetScreenInfo();
00135 
00136             // Then let current bottom position move one page down while printing the page.
00137             const StringDevice cli_Out(cli_ScreenInfo.GetSafeHeight() * (cli_ScreenInfo.GetSafeWidth() + 1), false);
00138             CLI_ASSERT(m_puiTextIt != NULL);
00139             if (m_puiTextIt != NULL) { m_uiText.PrintPage(*m_puiTextIt, cli_Out, true); }
00140 
00141             // Eventually clean out the screen and print out the computed string.
00142             m_cliLessLine.Reset();
00143             GetShell().GetStream(OUTPUT_STREAM).CleanScreen();
00144             GetShell().GetStream(OUTPUT_STREAM) << cli_Out.GetString();
00145             m_cliLessLine.Put(GetShell().GetStream(OUTPUT_STREAM), tk::String(10, ":"));
00146         }
