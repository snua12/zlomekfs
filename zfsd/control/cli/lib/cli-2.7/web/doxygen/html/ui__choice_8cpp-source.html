00039         static const ResourceString Choice2Str(
00040                 const int I_Choice,                             
00041                 const tk::Queue<ResourceString>& TK_Choices     
00042                 )
00043         {
00044             // Loop over the choice list until the number of elements to skip has been reached.
00045             tk::Queue<ResourceString>::Iterator it =  TK_Choices.GetIterator();
00046             for (   int i_ElementsToSkip = I_Choice;
00047                     (i_ElementsToSkip >= 0) && TK_Choices.IsValid(it);
00048                     i_ElementsToSkip --)
00049             {
00050                 if (i_ElementsToSkip == 0)
00051                 {
00052                     return TK_Choices.GetAt(it);
00053                 }
00054                 else
00055                 {
00056                     TK_Choices.MoveNext(it);
00057                 }
00058             }
00059 
00060             // Default to an empty string.
00061             return ResourceString();
00062         }

00076         const int Choice::GetChoice(void) const
00077         {
00078             const tk::String str_Line = Line::GetLine();
00079             int i_Choice = 0;
00080             tk::Queue<int> tk_Near(10);
00081             for (   tk::Queue<ResourceString>::Iterator it = m_tkChoices.GetIterator();
00082                     m_tkChoices.IsValid(it);
00083                     m_tkChoices.MoveNext(it), i_Choice++)
00084             {
00085                 const tk::String str_Choice = m_tkChoices.GetAt(it).GetString(m_eLang);
00086 
00087                 // Check whether the line matches the choice exactly.
00088                 if (str_Line == str_Choice)
00089                 {
00090                     return i_Choice;
00091                 }
00092 
00093                 // Check whether the line starts like the choice (case insensitive).
00094                 if (str_Line.ToUpper() == str_Choice.SubString(0, str_Line.GetLength()).ToUpper())
00095                 {
00096                     tk_Near.AddTail(i_Choice);
00097                 }
00098             }
00099 
00100             // If only one near result, then this one is the choice.
00101             if (tk_Near.GetCount() == 1)
00102             {
00103                 return tk_Near.GetHead();
00104             }
00105 
00106             // No matching choice.
00107             return -1;
00108         }

00134         {
00135             switch (E_KeyCode)

00140             case KEY_DOWN:

00143             case PAGE_UP:

00149             case ENTER:

00152                     Line::SetLine(GetstrChoice().GetString(m_eLang), true, true);

00157                     GetShell().Beep();

00159                 break;
00160             default:
00161                 Line::OnKey(E_KeyCode);

00166         void Choice::MoveChoice(const int I_Way, const unsigned int UI_Increment)
00167         {
00168             const int i_Increment = I_Way * (int) ((UI_Increment == 0) ? 1 : UI_Increment);
00169             const int i_Choice = GetChoice();
00170 
00171             if (i_Choice < 0)
00172             {
00173                 // Undefined choice: restore default.
00174                 GetShell().Beep();
00175                 ResetToDefault();
00176             }
00177             else if (i_Choice + i_Increment < 0)
00178             {
00179                 // Top of choice list.
00180                 // Possibly beep.
00181                 if (i_Choice <= 0)
00182                 {
00183                     GetShell().Beep();
00184                 }
00185                 // Reprint.
00186                 Line::SetLine(Choice2Str(0, m_tkChoices).GetString(m_eLang), false, true);
00187             }
00188             else if (i_Choice + i_Increment >= (int) m_tkChoices.GetCount())
00189             {
00190                 // Bottom of choice list.
00191                 // Possibly beep.
00192                 if (i_Choice >= ((int) m_tkChoices.GetCount()) - 1)
00193                 {
00194                     GetShell().Beep();
00195                 }
00196                 // Reprint.
00197                 Line::SetLine(Choice2Str(m_tkChoices.GetCount() - 1, m_tkChoices).GetString(m_eLang), false, true);
00198             }
00199             else
00200             {
00201                 // Move to next element.
00202                 Line::SetLine(Choice2Str(GetChoice() + i_Increment, m_tkChoices).GetString(m_eLang), false, true);
00203             }
00204         }
