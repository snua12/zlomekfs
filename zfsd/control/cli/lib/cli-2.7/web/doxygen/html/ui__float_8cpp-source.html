00085                 if (GetFloat() >= m_dMaxValue)

00087                     // Upper bound already reached

00090                 // Ensure max value.
00091                 Line::SetLine(Float2Str(m_dMaxValue), false, false);
00092                 break;
00093             case PAGE_DOWN:
00094                 if (GetFloat() <= m_dMinValue)
00095                 {
00096                     // Lower bound already reached
00097                     GetShell().Beep();
00098                 }
00099                 // Ensure min value.
00100                 Line::SetLine(Float2Str(m_dMinValue), false, false);
00101                 break;
00102             case ENTER:
00103                 if ((GetFloat() >= m_dMinValue) && (GetFloat() <= m_dMaxValue))
00104                 {
00105                     // Reprint understood value to avoid confusions from the user.
00106                     double d_Value = GetFloat();
00107                     Line::SetLine(Float2Str(d_Value), true, false);
00108                     UI::Finish(true);
00109                 }
00110                 else
00111                 {
00112                     GetShell().Beep();
00113                 }
00114                 break;
00115             default:
00116                 Line::OnKey(E_KeyCode);
00117                 break;
00118             }
00119         }
