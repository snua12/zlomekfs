00126         const bool Text::LineUp(TextIterator& it) const
00127         {
00128             if (m_tkLines.IsValid(it.m_tkTopLine))
00129             {
00130                 // Check whether the current top character position is not aligned to the beginning of a line.
00131                 if (it.m_uiTopChar >= it.m_cliScreenInfo.GetSafeWidth())
00132                 {
00133                     it.m_uiTopChar -= it.m_cliScreenInfo.GetSafeWidth();
00134                     it.m_bBottomIsUpToDate = false; // Bottom position is out of date until PrintPage() is called.
00135                     return true;
00136                 }
00137 
00138                 // Try to decrement the current top line position.
00139                 if (m_tkLines.MovePrevious(it.m_tkTopLine))
00140                 {
00141                     CLI_ASSERT(it.m_cliScreenInfo.GetSafeWidth() > 0);
00142                     if (it.m_cliScreenInfo.GetSafeWidth() <= 0)
00143                     {
00144                         // Should not happen unless there is a bug in OutputDevice::ScreenDimension.
00145                         // If ever, let's start at the beginning of the upper line.
00146                         it.m_uiTopChar = 0;
00147                     }
00148                     else
00149                     {
00150                         it.m_uiTopChar = m_tkLines.GetAt(it.m_tkTopLine).GetLength();
00151                         if (m_tkLines.GetAt(it.m_tkTopLine).GetLength() % it.m_cliScreenInfo.GetSafeWidth() == 0)
00152                         {
00153                             it.m_uiTopChar -= it.m_cliScreenInfo.GetSafeWidth();
00154                         }
00155                         else
00156                         {
00157                             it.m_uiTopChar -= m_tkLines.GetAt(it.m_tkTopLine).GetLength() % it.m_cliScreenInfo.GetSafeWidth();
00158                         }
00159                     }
00160                     it.m_bBottomIsUpToDate = false; // Bottom position is out of date until PrintPage() is called.
00161 
00162                     return true;
00163                 }
00164                 else
00165                 {
00166                     // Restore position to beginning.
00167                     CLI_ASSERT(it.m_uiTopChar == 0);

00265         void Text::PrintBottomLine(TextIterator& it, const OutputDevice& CLI_Out) const
00266         {
00267             for (   unsigned int ui_CharCount = 0;
00268                     m_tkLines.IsValid(it.m_tkBottomLine);  // While there are still lines to display.
00269                     ui_CharCount ++)
00270             {
00271                 const tk::String tk_Line = m_tkLines.GetAt(it.m_tkBottomLine);
00272 
00273                 if (it.m_uiBottomChar >= tk_Line.GetLength())
00274                 {
00275                     // End of line.
00276                     CLI_ASSERT(it.m_cliScreenInfo.GetSafeWidth() > 0);
00277                     if ((! it.m_cliScreenInfo.GetbWrapLines())
00278                         // When line wrapping is on, a line ending at the right end of the screen should not cause an empty line below.
00279                         // Thus, print a carriage return only in the following cases:
00280                         || (it.m_uiBottomChar == 0)                                         // Regular empty line.
00281                         || (it.m_cliScreenInfo.GetSafeWidth() <= 0)                         // (Avoid invalid modulo, should never occure however)
00282                         || (it.m_uiBottomChar % it.m_cliScreenInfo.GetSafeWidth() != 0))    // Line not ending at the right end the screen.
00283                     {
00284                         CLI_Out << endl;
00285                     }
00286                     m_tkLines.MoveNext(it.m_tkBottomLine);
00287                     it.m_uiBottomChar = 0;
00288                     break;
00289                 }
00290                 else if (ui_CharCount >= it.m_cliScreenInfo.GetSafeWidth())
00291                 {
00292                     // Out of line.
00293                     if (! it.m_cliScreenInfo.GetbWrapLines())
00294                     {
00295                         CLI_Out << endl;
00296                     }
00297                     break;
00298                 }
00299                 else
00300                 {
00301                     // Regular character display.
00302                     CLI_Out << tk_Line.GetChar(it.m_uiBottomChar);
00303                     it.m_uiBottomChar ++;
00304                 }
00305             }
00306         }
