00982 void Shell::OnHelp(const bool B_PromptMenu, const bool B_Completion)
00983 {
00984     if (! m_qMenus.IsEmpty())
00985     {
00986         // Parse the command line.
00987         CommandLine cli_CommandLine;
00988         if (cli_CommandLine.Parse(*m_qMenus.GetTail(), m_cliCmdLine.GetLeft(), false))
00989         {
00990             // Research matching elements.
00991             Element::List cli_Exact(MAX_WORDS_PER_NODE), cli_Elements(MAX_WORDS_PER_NODE);
00992             if (! cli_CommandLine.GetLastElement().FindElements(
00993                     cli_Exact, cli_Elements, cli_CommandLine.GetLastWord()))
00994             {
00995                 return;
00996             }
00997 
00998             // Initialize completion so far.
00999             tk::String str_CompletionSoFar(MAX_WORD_LENGTH);
01000             // Determine whether help should be printed, and completion activated.
01001             bool b_PrintHelpList = false;
01002 
01003             if (! B_Completion)
01004             {
01005                 // Just print help.
01006                 b_PrintHelpList = true;
01007             }
01008             else
01009             {
01010                 // Print help and do completion depending on the possible nodes.
01011                 if (cli_Elements.IsEmpty())
01012                 {
01013                     // No matching element.
01014                     Beep();
01015                 }
01016                 else if ((cli_Elements.GetCount() == 1)
01017                         && (! dynamic_cast<const Endl*>(cli_Elements.GetHead()))
01018                         && (! dynamic_cast<const Param*>(cli_Elements.GetHead())))
01019                 {
01020                     // Straight forward completion.
01021                     m_cliCmdLine.Delete(GetStream(ECHO_STREAM), - cli_CommandLine.GetNumBackspacesForCompletion());
01022                     m_cliCmdLine.Put(GetStream(ECHO_STREAM), cli_Elements.GetHead()->GetKeyword());
01023                     m_cliCmdLine.Put(GetStream(ECHO_STREAM), ' ');
01024                 }
01025                 else
01026                 {
01027                     // Look for completion so far.
01028                     bool b_InitCompletionSoFar = true;
01029                     for (   Element::List::Iterator it = cli_Elements.GetIterator();
01030                             cli_Elements.IsValid(it);
01031                             cli_Elements.MoveNext(it))
01032                     {
01033                         if (dynamic_cast<const Param*>(cli_Elements.GetAt(it))
01034                             || dynamic_cast<const Endl*>(cli_Elements.GetAt(it)))
01035                         {
01036                             // No completion for parameters and end of lines.
01037                             str_CompletionSoFar.Set("");
01038                             break;
01039                         }
01040                         else if (const Element* const pcli_Element = dynamic_cast<const Element*>(cli_Elements.GetAt(it)))
01041                         {
01042                             if (b_InitCompletionSoFar)
01043                             {
01044                                 // First initialization of str_CompletionSoFar.
01045                                 if (! str_CompletionSoFar.Set(pcli_Element->GetKeyword()))
01046                                     GetTraces().Trace(INTERNAL_ERROR) << "Shell::OnHelp(): Could not store completion so far" << endl;
01047                                 b_InitCompletionSoFar = false;
01048                             }
01049                             else
01050                             {
01051                                 // Remove last character of str_CompletionSoFar till it matches the current keyword.
01052                                 while ( (! str_CompletionSoFar.IsEmpty())
01053                                         && (pcli_Element->GetKeyword().SubString(0, str_CompletionSoFar.GetLength()) != str_CompletionSoFar))
01054                                 {
01055                                     if (! str_CompletionSoFar.Set(str_CompletionSoFar.SubString(0, str_CompletionSoFar.GetLength() - 1)))
01056                                         GetTraces().Trace(INTERNAL_ERROR) << "Shell::OnHelp(): Could not store completion so far" << endl;
01057                                 }
01058                             }
01059                         }
01060                     }
01061 
01062                     // When no more completion so far available, print help.
01063                     if ((cli_CommandLine.GetLastWord() == NULL)
01064                         || (str_CompletionSoFar == cli_CommandLine.GetLastWord()))
01065                     {
01066                         b_PrintHelpList = true;
01067                     }
01068                 }
01069             }
01070 
01071             // Help display.
01072             if (b_PrintHelpList)
01073             {
01074                 // First of all, sort help elements on keywords.
01075                 class _ { public:
01076                     static const int cmp(const Element* const& PCLI_1, const Element* const& PCLI_2)
01077                     {
01078                         if (PCLI_1->GetKeyword() < PCLI_2->GetKeyword())
01079                             return 1;
01080                         if (PCLI_1->GetKeyword() > PCLI_2->GetKeyword())
01081                             return -1;
01082                         return 0;
01083                     }
01084                 };
01085                 cli_Elements.Sort(_::cmp);
01086 
01087                 if (B_PromptMenu)
01088                 {
01089                     m_cliCmdLine.NextLine(GetStream(ECHO_STREAM));
01090                 }
01091 
01092                 // For each matching keyword...
01093                 for (   Element::List::Iterator it = cli_Elements.GetIterator();
01094                         cli_Elements.IsValid(it);
01095                         cli_Elements.MoveNext(it))
01096                 {
01097                     // Print the corresponding help.
01098                     PrintHelp(*cli_Elements.GetAt(it));
01099                 }
01100 
01101                 if (B_PromptMenu)
01102                 {
01103                     PromptMenu();
01104                 }
01105             }
01106             // Completion so far.
01107             else if (B_Completion && (! str_CompletionSoFar.IsEmpty()))
01108             {
01109                 m_cliCmdLine.Delete(GetStream(ECHO_STREAM), - cli_CommandLine.GetNumBackspacesForCompletion());

01111             }
01112         }
01113         else
01114         {
01115             if (B_PromptMenu)
01116             {
01117                 m_cliCmdLine.NextLine(GetStream(ECHO_STREAM));
