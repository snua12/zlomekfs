00087         void More::OnKey(const KEY E_KeyCode)
00088         {
00089             // Ensure m_puiTextIt is valid.
00090             CLI_ASSERT(m_puiTextIt != NULL);
00091             if (m_puiTextIt == NULL) { Quit(); }
00092             else {
00093                 switch (E_KeyCode)
00094                 {
00095                 case KEY_END:
00096                     HideMoreMessage();
00097                     // We are already at the bottom of a page, no need to optimize display so far, let it progress fluently.
00098                     // Optimize line by line at least.
00099                     for (bool b_LineDown = true; b_LineDown; )
00100                     {
00101                         const OutputDevice::ScreenInfo cli_ScreenInfo = GetShell().GetStream(OUTPUT_STREAM).GetScreenInfo();
00102                         const StringDevice cli_Out(cli_ScreenInfo.GetSafeHeight() * (cli_ScreenInfo.GetSafeWidth() + 1), false);
00103                         b_LineDown = m_uiText.LineDown(*m_puiTextIt, & cli_Out);
00104                         GetShell().GetStream(OUTPUT_STREAM) << cli_Out.GetString();
00105                     }
00106                     ShowMoreMessage();
00107                     break;
00108                 case PAGE_DOWN:
00109                 case SPACE:
00110                     // Print one more page
00111                     HideMoreMessage();
00112                     do {
00113                         // Output lines to a buffer output device.
00114                         const OutputDevice::ScreenInfo cli_ScreenInfo = GetShell().GetStream(OUTPUT_STREAM).GetScreenInfo();
00115                         const StringDevice cli_Out(cli_ScreenInfo.GetSafeHeight() * (cli_ScreenInfo.GetSafeWidth() + 1), false);
00116                         m_uiText.PageDown(*m_puiTextIt, & cli_Out);
00117                         // Display in one call in order to optimize display.
00118                         GetShell().GetStream(OUTPUT_STREAM) << cli_Out.GetString();
00119                     } while(0);
00120                     ShowMoreMessage();
00121                     break;
00122                 case KEY_DOWN:
00123                 case ENTER:
00124                     // Print one more line
00125                     HideMoreMessage();
00126                     m_uiText.LineDown(*m_puiTextIt, & GetShell().GetStream(OUTPUT_STREAM));
00127                     ShowMoreMessage();
00128                     break;
00129                 case KEY_q:
00130                 case KEY_Q:
00131                 case ESCAPE:
00132                 case BREAK:
00133                 case LOGOUT:
00134                 case NULL_KEY:
00135                     // Stop display
00136                     Quit();
00137                     break;
00138                 default:
00139                     // Non managed character. Beep.
00140                     GetShell().Beep();
00141                     break;
00142                 }
00143             }
00144         }

00146         void More::ShowMoreMessage(void)
00147         {
00148             CLI_ASSERT(m_puiTextIt != NULL);
00149             if (m_puiTextIt != NULL)
00150             {
00151                 m_cliMoreLine.Reset();
00152                 TextIterator tmp(*m_puiTextIt);
00153                 if (m_uiText.LineDown(tmp, NULL))
00154                 {
00155                     // Still lines to display.
00156                     const ResourceString cli_MoreMessage = ResourceString()
00157                         .SetString(ResourceString::LANG_EN, "--- More ---")
00158                         .SetString(ResourceString::LANG_FR, "--- Plus ---");
00159                     m_cliMoreLine.Put(GetShell().GetStream(OUTPUT_STREAM), cli_MoreMessage.GetString(GetShell().GetLang()));
00160                     return;
00161                 }
00162             }
00163 
00164             // If the input text is done (or something wrong occured), terminate the UI execution.
00165             Quit();
00166         }
