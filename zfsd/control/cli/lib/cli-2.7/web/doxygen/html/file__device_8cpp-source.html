00122         // While there are still characters to read.
00123         while ((! m_tkInputBuffer.IsEmpty()) || (! feof(m_pfFile)))
00124         {
00125             // Check the input buffer
00126             if (m_tkInputBuffer.IsEmpty())

00128                 if ((m_pfFile != NULL) && (! feof(m_pfFile)))

00134                         for (size_t ui=0; (ui<ui_Bytes) && (ui<sizeof(str_Buffer)); ui++)

00137                         }
00138                     }
00139                     else
00140                     {

00144                         return NULL_KEY;
00145                     }
00146                 }
00147             }
00148 
00149             // Read the next character.
00150             if (! m_tkInputBuffer.IsEmpty())
00151             {
00152                 char c_Char = m_tkInputBuffer.RemoveHead();
00153 
00154                 // Next line/column management & special character management.
00155                 switch (c_Char)
00156                 {
00157                 case '\n':
00158                     m_iCurrentLine = m_iNextLine;       m_iNextLine ++;
00159                     m_iCurrentColumn = m_iNextColumn;   m_iNextColumn = 1;
00160                     break;
00161                 case '?':
00162                     if ((m_iNextLine == m_iCurrentLine) && (m_iNextColumn == m_iCurrentColumn) && (! m_bEnableSpecialCharacters))
00163                     {
00164                         // Income of the effective '?' character, after it has been escaped (see below).
00165                         // This time, step over.
00166                         m_iNextColumn ++;
00167                     }
00168                     else if (! m_bEnableSpecialCharacters)
00169                     {
00170                         // Escape the '?' character. Post it back. Do not step over yet.
00171                         m_iCurrentLine = m_iNextLine;
00172                         m_iCurrentColumn = m_iNextColumn;
00173                         c_Char = '\\';
00174                         m_tkInputBuffer.AddHead('?');
00175                     }
00176                     else
00177                     {
00178                         // Just step over.
00179                         m_iCurrentLine = m_iNextLine;
00180                         m_iCurrentColumn = m_iNextColumn ++;
00181                     }
00182                     break;
00183                 case '\t':
00184                     if (! m_bEnableSpecialCharacters)
00185                     {
00186                         // Change tabs into spaces.
00187                         c_Char = ' ';
00188                     }
00189                     // Step over.
00190                     m_iCurrentLine = m_iNextLine;
00191                     m_iCurrentColumn = m_iNextColumn ++;
00192                 default:
00193                     // Just step over.
00194                     m_iCurrentLine = m_iNextLine;
00195                     m_iCurrentColumn = m_iNextColumn ++;
00196                     break;
00197                 }
00198 
00199                 // Character analysis.
00200                 KEY e_Key = IODevice::GetKey(c_Char);
00201                 if (e_Key == NULL_KEY)
00202                 {

00205                         .SetString(ResourceString::LANG_FR, ResourceString::Concat("Erreur de lecture du fichier d'entrée '", m_strFileName, "'"));
00206                 }
