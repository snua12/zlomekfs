00050 const bool CmdLineHistory::Push(const CmdLineEdition& CLI_Line)
00051 {
00052     if (// Check the line is not empty.
00053         (! CLI_Line.GetLine().IsEmpty())
00054         // Check it is not the same as the previous one.
00055         && ((m_qHistory.GetCount() <= 1) || (CLI_Line.GetLine() != GetLine(1).GetLine())))
00056     {
00057         // Limit to HISTORY_STACK_SIZE.
00058         while (m_qHistory.GetCount() > m_uiStackSize)
00059         {
00060             m_qHistory.RemoveTail();
00061         }
00062 
00063         // Ensure there is at least one element.
00064         if (m_qHistory.IsEmpty())
00065         {
00066             if (! m_qHistory.AddTail(CmdLineEdition()))
00067             {
00068                 return false;
00069             }
00070         }
00071 
00072         // Backup the current line.
00073         m_qHistory.GetHead() = CLI_Line;
00074 
00075         // Add a new empty element.
00076         m_qHistory.AddHead(CmdLineEdition());
00077     }
00078 
00079     m_uiNavigationIndex = 0;
00080     m_bNavigation = false;
00081 
00082     return true;
